Class ScriptPetUnitAI

This is a class to access and control the pet's AI.

Examples
-- Example
-- Pet AI designation

Server.SetPetAI(
        21, -- Character ID to be pet
        function(pet,ai,event)

                -- Executed when registering for the first PetAI
                if(event == AI_INIT) then
                        ai.customData.timer = 1;
                end

                -- Every 2 seconds
                if (event == AI_UPDATE) then
                        -- Follow the owner as far as 200
                        -- Teleports to the owner's position as far as 400
                        -- ai.SetFollowMaster(true, 200, 400)

                        -- If there is no target, follow the owner
                        if (ai.GetTargetUnit() == nil) then
                                ai.SetFollowMaster(true, 200, 400)
                        end

                        -- Basic 100, 200
                        -- ai.SetFollowMaster(true)

                        -- Target the nearest enemy unit
                        ai.SetNearTarget(2, 200) 

                        -- If the pet's target exists, use the skill
                        -- Skills are basically fired towards the target
                        if (ai.GetTargetUnit() ~=nil) then
                                -- When the target is set, it stops following
                                ai.SetFollowMaster(false)
                                ai.StopMove()

                                -- When target is decided, attack while following target
                                ai.MoveToPosition(ai.GetTargetUnit().x,ai.GetTargetUnit().y)
                                -- Add a buff to the owner
                                ai.AddMasterBuff(15)
                                -- Fire in the direction of the target
                                ai.UseSkill(23) 
                                -- Fire from the pet's master position to the target
                                ai.UseSkill(
                                22
                                ,nil
                                ,Point(ai.GetMasterUnit().x,ai.GetMasterUnit().y))
                        end

                        -- Timer using custom data Once around 6 seconds, drop items are acquired
                        ai.customData.timer = ai.customData.timer + 1;
                        if (ai.customData.timer == 3) then
                                -- Drop items within a radius of 100
                                ai.AcquireNearDropItem(100)
                                ai.customData.timer = 0;
                        end
                end
        end
)
