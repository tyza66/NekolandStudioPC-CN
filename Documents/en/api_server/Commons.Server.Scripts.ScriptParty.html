KickParty(ScriptRoomPlayer)

Kick out the party

Declaration
public bool KickParty(ScriptRoomPlayer player)
Parameters
Type Name Description
ScriptRoomPlayer player

the player to kick out

Returns
Type Description
System.Boolean

Success or not

Examples
-- Force all party members except me to leave the party

-- Check if there is a party
if unit.party == nil then
    unit.SendCenterLabel('The party you joined does not exist')
    return;
end

-- List all party members, get the number of party members
local players = unit.party.players
local playerCount = #players

-- Exception handling when the number of party members is less than 1 (if I am alone)
if playerCount <= 1 then
    unit.SendCenterLabel('There are no other users at the party')
    return;
end

local outCount = 0
-- Tour the number of party members and leave each party member
for i = 1, playerCount do
    if players[i] ~= unit.player and players[i] ~= nil then
        unit.party.KickParty(players[i])
        outCount = outCount + 1
    end
end

-- Notification display
unit.SendCenterLabel(outCount .. "users were sent from the party")
