Examples
-- Parse a JSON string containing name and age information and print the data

local jsonString = '{"name":"cat","age":50}'
local jsonTable = Utility.JSONParse(jsonString)

local name = jsonTable["name"]
local age = jsonTable["age"]

print('name=' .. name .. ' age=' .. age)

-- Output: name=cat age=50
