Property Value
Type Description
ScriptEventPublisher
Examples
-- Notify all players in the field when a drop item is added to the field
-- Note: This example can work recursively

-- Callback when drop item is actually dropped
function onJoinDropItem(field, dropItem)
    if dropItem == nil then
        return
    end

    local itemData = Server.GetItem(dropItem.titem.dataID)
    if itemData ~= nil then
        field.SendCenterLabel(itemData.name .. "item fell to the ground!")
    end
end


-- Function to register callback in target field
function registerCallback()
    local field = Server.GetField(6)

    -- Callback registration if field is created
    if field ~= nil then
        field.onJoinDropItem.Add(onJoinDropItem)
    else
        -- If not created, wait 1 second again and try to register the callback
        Server.RunLater(registerCallback, 1)
    end
end


-- Execute the registration function after 1 second
-- (field may not have been created yet at the time the script is executed)
Server.RunLater(registerCallback, 1)

Examples
-- When entering a specific map, print out the name and send the entire server message

function onJoinField()
    function onJoinField_1(field, unit)
        unit.SendCenterLabel(field.name) -- Display the field name briefly to the unit.
        if #field.playerUnits > 10 then -- # Means the length of the table.
            Server.SendSay("More than 11 users gathered in" .. field.name)
        end
    end

    local map = Server.GetField(1)
    if map ~= nil then
        -- When entering field 1, onJoinField_1 function is called.
        map.onJoinField.Add(onJoinField_1)
    end
end

-- After 1 second, execute the onJoinField function.
-- The reason to run after 1 second is to prevent referencing nil because the map may not have been created after server execution.
Server.RunLater(onJoinField, 1)
