Class ScriptScrollPanel

Class to display the scrolled panel

Inheritance
System.Object
ScriptBaseControl
ScriptControl
ScriptScrollPanel
Namespace: Game.Scripts
Assembly: Assembly-CSharp.dll
Syntax
[MoonSharpUserData]
public class ScriptScrollPanel : ScriptControl
Examples
-- Put in a page scroll panel created with the layout manager
panel = Panel(Rect(0,0,Client.width,Client.height))
panel.anchor = 4
panel.pivotX = 0.5
panel.pivotY = 0.5
panel.x = 0
panel.y = 0
panel.masked = true
panel.LoadPage("page_0")

scrollPanel = ScrollPanel(Rect(300,300,100,100))
scrollPanel.masked = true
scrollPanel.content.width = panel.width
scrollPanel.content.height = panel.height
scrollPanel.content.AddChild(panel)
scrollPanel.horizontal = true
scrollPanel.vertical = true 
scrollPanel.showHorizontalScrollbar = true
scrollPanel.showVerticalScrollbar = true
