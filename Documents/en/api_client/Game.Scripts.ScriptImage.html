function CoolTime(deltaTime)
    leftTime = leftTime - deltaTime
    if leftTime <= 0 then
        leftTime = coolTime
        print("end")
        img.fillAmount = 1
         Client.onTick.Remove(CoolTime)
        return
    end
    img.fillAmount = 1 - (leftTime / coolTime)
end

Client.onTick.Add(CoolTime)
