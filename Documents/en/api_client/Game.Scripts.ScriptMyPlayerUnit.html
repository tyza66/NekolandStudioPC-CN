Class ScriptMyPlayerUnit

This class represents the current client's player unit.

Inheritance
System.Object
ScriptUnit
ScriptMyPlayerUnit
Namespace: Game.Scripts
Assembly: Assembly-CSharp.dll
Syntax
[MoonSharpUserData]
public class ScriptMyPlayerUnit : ScriptUnit
Examples
-- Use the skill on the nearest enemy unit
me = Client.myPlayerUnit
nearEnemyUnit = Client.field.FindNearUnit(me.x, me.y, 200, 2, me)
if nearEnemyUnit ~= nil then
 me.UseSkill(22,Point(nearEnemyUnit.x - me.x,nearEnemyUnit.y - me.y))
end

-- Move to the nearest unit location
me = Client.myPlayerUnit
nearEnemyUnit = Client.field.FindNearUnit(me.x, me.y, 200, 2, me)
if nearEnemyUnit ~= nil then
 me.SetTarget(nearEnemyUnit)
end
