DOKill()

이 컨트롤이 수행중인 모든 애니메이션을 종료합니다 (DO .. 시리즈 함수)

선언
public void DOKill()
예제
-- 패널 컨트롤을 생성하여 이동, 회전, 사이즈 조정을 시작한 뒤
-- 0.5초 후 DOKill을 이용하여 모든 행동을 취소합니다.
local panel = Panel(Rect(100, 100, 100, 100))

-- Transform 조정
panel.DOMove(Point(100, 200), 1);
panel.DOScale(Point(2, 2), 1);
panel.DORotate(180, 1);

Client.RunLater(
    function()
        panel.DOKill();

    end, 0.5);
